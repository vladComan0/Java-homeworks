/*Write a Frame application that will display in a panel your name 
and group. The text will be blue and centered both on the horizontal 
and vertical, given the dimension of the panel. 
Vlad Coman, Group 2024
18.05.2020
*/
package lab10;
import java.awt.Canvas;
import java.awt.Color;
import java.awt.FlowLayout;
import java.awt.Font;
import java.awt.Frame;
import java.awt.Graphics;
import java.awt.GridBagConstraints;
import java.awt.Label;
import java.awt.Panel;
import java.awt.event.*;
import java.awt.Dimension;
class MyFrame extends Frame {
	private static final long serialVersionUID = 1L;
	MyFrame(String title){
	super(title);
	setLayout(new FlowLayout());
	Panel panel = new Panel();
	Label label = new Label("Coman Vladut");
	label.setFont(new Font("Times New Roman", 1, 20));
	panel.add(label);
	this.add(panel);
	this.setLocationRelativeTo(null);
	this.addWindowListener(new ExitListener());
	setSize(300, 300);
	setVisible(true);
	}
	}
class ExitListener extends WindowAdapter {//closing with adapter class
	 @Override
	 public void windowClosing(WindowEvent event) {
	 System.exit(0);
	}
}
public class Main{
public static void main(String args[ ]) {
	MyFrame frame = new MyFrame("Name App");
}
}

a circle tangent to smaller side of the canvas
Vlad Coman, Group 2024
18.05.2020
*/
package lab10;
import java.awt.*;
import java.awt.event.*;
class MyFrame extends Frame {
	private static final long serialVersionUID = 1L;
	MyFrame(String title){
	super(title);
	MyCanvas canvas=new MyCanvas(200,100);
	add(canvas);
	setLayout(new FlowLayout());
	this.addWindowListener(new ExitListener());
	setSize(300, 300);
	setVisible(true);
	}
	}
class ExitListener extends WindowAdapter {//closing with adapter class
	 @Override
	 public void windowClosing(WindowEvent event) {
	 System.exit(0);
	}
}
class MyCanvas extends Canvas{
	private static final long serialVersionUID = 1L;
	Graphics g;
	int x, y;
	MyCanvas(int x, int y){
		this.x=x;
		this.y=y;
		this.setSize(new Dimension(x, y));
		this.setBackground(Color.blue);
	}
	@Override
	public void paint(Graphics g) {
		g.setColor(Color.white);
		if(x>y) {
		g.fillOval(0, 10, 50, 50);
		}
		else{
			g.fillOval(10, 0, 50, 50);
		}
	}
}
public class Main{
public static void main(String args[ ]) {
	MyFrame frame = new MyFrame("Circle Application");
}
}

	setSize(490, 530);
	setVisible(true);
	}
	}
class ExitListener extends WindowAdapter {//closing with adapter class
	 @Override
	 public void windowClosing(WindowEvent event) {
	 System.exit(0);
	}
}
class MyCanvas extends Canvas{
	private static final long serialVersionUID = 1L;
	public MyCanvas() {
		this.setSize(new Dimension(490, 490));
		this.setBackground(new Color(0, 0, 0));
	}
	public void paint(Graphics g) {
		int red=255;
		int green=0;
		int blue=0;
		for(int i=0;i<60;i++) {
			g.setColor(new Color(red,green,blue));
			g.fillArc(150, 150, 200, 200, i, 1);
			green+=4;
			}
		for(int i=60;i<120;i++) {
			g.setColor(new Color(red,green,blue));
			g.fillArc(150, 150, 200, 200, i, 1);
			red-=4;
		}
		for(int i=120;i<180;i++) {
			g.setColor(new Color(red,green,blue));
			g.fillArc(150, 150, 200, 200, i, 1);
			green-=2;
			blue+=4;
		}
		for(int i=180;i<240;i++) {
			g.setColor(new Color(red,green,blue));
			g.fillArc(150, 150, 200, 200, i, 1);
			green-=2;
		}
		for(int i=240;i<300;i++) {
			g.setColor(new Color(red,green,blue));
			g.fillArc(150, 150, 200, 200, i, 1);
			green+=2;
			red+=2;
		}
		for(int i=300;i<360;i++) {
			g.setColor(new Color(red,green,blue));
			g.fillArc(150, 150, 200, 200, i, 1);
			green-=2;
			red+=2;
			blue-=1;
		}
		}
}
public class Main{

	MyFrame frame = new MyFrame("Color Palette");

/*Implement a Frame application that fills a visual container with 
circles having randomly generated radii. The circles are tangent and 
have random colors.
Vlad Coman, Group 2024
18.05.2020
*/
package lab10;
import java.awt.*;
import java.awt.event.*;
import java.util.Random;
class MyFrame extends Frame {
	private static final long serialVersionUID = 1L;
	MyFrame(String title){
	super(title);
	MyCanvas canvas=new MyCanvas();
	add(canvas);
	setLayout(new FlowLayout());
	this.addWindowListener(new ExitListener());
	setSize(600, 530);
	setVisible(true);
	}
	}
class ExitListener extends WindowAdapter {//closing with adapter class
	 @Override
	 public void windowClosing(WindowEvent event) {
	 System.exit(0);
	}
}
class MyCanvas extends Canvas{
	int x, y;
	private static final long serialVersionUID = 1L;
	public MyCanvas() {
		this.setSize(new Dimension(400,400));
		this.setBackground(new Color(200,200,200));
	}
	public void paint(Graphics g) {
		Random rand=new Random();
		int radius=rand.nextInt(400);
		int red, green, blue;
		for(int i=0;i<400;i+=radius)
			for(int j=0;j<400;j+=radius) {
			red=rand.nextInt(255);
			green=rand.nextInt(255);
			blue=rand.nextInt(255);
			g.setColor(new Color(red,green,blue));
			g.fillOval(i, j, radius, radius);
			}
		}
}
public class Main{
public static void main(String args[]) {
	MyFrame frame = new MyFrame("Color Palette");
}
