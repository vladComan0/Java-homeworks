/*Write a class named Methods having 3 methods: a method with a variable number of integer parameters, which
returns the arithmetic mean of the parameters; another method in which will be defined a one-dimensional 
array of n double-type numbers (n being the parameter of the method and which will be read in main () from
the keyboard), its elements will receive randomly generated values and the method will return this array; a
static method that receives as a parameter a character array, converts the respective characters into uppercase
letters and forms a String object from this array, which will be returned by the function. From the main ()
method, defined in another class, call the methods of the Methods class and display the results on the screen
(the call values for the first method and the character string for the 3rd method will be defined in main ()).
Vlad Coman, Group 2024
03.04.2020
*/
import java.util.Random;
import java.util.Scanner;
class Methods{
	double Method1(int... value) {
		int s=0;
		for(int element:value)
			s+=element;
		return (double)s/value.length;
	}
	double[] Method2(int n) {
		double []arr=new double[n];
		Random rand=new Random();
		int x;
		for(int i=0;i<n;i++) {
			x=rand.nextInt(100);
			arr[i]=x+rand.nextDouble();
		}
		return arr;
	}
	static String Method3(char []arr) {
		String str;
		for(int i=0;i<arr.length;i++){
			arr[i]=Character.toUpperCase(arr[i]);
		}
		str=new String(arr);
		return str;
	}
}
public class MyClass {
	private static Scanner cin;
	public static void main(String []args) {
		Methods ob1=new Methods();
		cin=new Scanner(System.in);

		n=cin.nextInt();
		arr=new int[n];
		System.out.println("Introduceti valori pentru cele "+n+" elemente");
		for(int i=0;i<n;i++)
			arr[i]=cin.nextInt();
		System.out.println("Media aritmetica este: "+ob1.Method1(arr));
		double []arr2=new double[n];
		arr2=ob1.Method2(n);
		System.out.println("Array-ul generat random este: ");
		for(int i=0;i<n;i++)
		System.out.println(arr2[i]);
		System.out.println("Introduceti numarul de litere: ");
		int nr;
		nr=cin.nextInt();
		char []arr3=new char[nr];
		System.out.println("Introduceti array-ul: ");
		for(int i=0;i<nr;i++)
			arr3[i]=cin.next().charAt(0);
		System.out.println("String-ul cu uppercase este: "+Methods.Method3(arr3));
	}
}

/*Write a Java class that models a matrix of integer values. The dimensions and the array of elements are private
attributes and are controlled using appropriate setter-getter methods. Write the methods for displaying the
matrix, for determining and returning the number of 9 adjacent elements cells that don’t differ with more than
5% from a threshold value.
Vlad Coman, Group 2024
03.04.2020
*/
import java.util.Scanner;
class Matrics{
	private int m, n;
	private int arr[][];
	public int[][] getArr() {
		return arr;
	}
	public void setArr(int arr[][]) {
		this.arr = arr;
	}
	public int getN() {
		return n;
	}
	public void setN(int n) {
		this.n = n;
	}
	public int getM() {
		return m;
	}
	public void setM(int m) {
		this.m = m;
	}
	public void display(int m, int n, int [][]arr) {
		for(int i=0;i<m;i++) {
			for(int j=0;j<n;j++) 
				System.out.print(arr[i][j]+" ");
		System.out.println(" ");
			}
	}
	public int adjacent(int m, int n, int [][]arr, int th) {
		float k=5/(float)100*th;
		int nr=0;
		for(int i=0;i<m;i++)
			for(int j=0;j<n;j++)
				if(arr[i][j]>=k || arr[i][j]<k) 
					nr++;
		return nr;
	}
}
public class MyClass {
	private static Scanner cin;
	public static void main(String []args) {
		cin=new Scanner(System.in);
		int m, n;
		System.out.println("Introduceti valorile pentru m si n: ");
		System.out.println("m= ");
		m=cin.nextInt();
		System.out.println("n= ");
		n=cin.nextInt();
		int [][]arr=new int[m][n];
		System.out.println("Introduceti elementele matricii: ");
		for(int i=0;i<m;i++)
			for(int j=0;j<n;j++)
				arr[i][j]=cin.nextInt();
		Matrics ob1=new Matrics();
		ob1.setArr(arr);
		ob1.setM(m);
		ob1.setN(n);
		ob1.display(m, n, arr);
		System.out.println("Numarul de elemente adiacente care nu difera cu 5% fata de 10 este: "+ob1.adjacent(m,n,arr,10));
	}
}
